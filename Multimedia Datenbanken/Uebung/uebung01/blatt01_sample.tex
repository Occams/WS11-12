\section{Aufgabe 1}
Mediendatentypen:
	\begin{itemize}
		\item Text 
		\item Bild 
		\item Video 
		\item Audio 
		\item Composed Media (z.B. Video mit Ton)
	\end{itemize}
	
\textbf{Text}
	\begin{itemize}
		\item diskretes Medium 
		\item Datengröße eher klein, daher keine Kompression notwendig
		\item Zeichenkodierung: ASCII, UTF-8
		\item Struktur 
		\item Layout
	\end{itemize}
	
	\textbf{Audio}
	\begin{itemize}
		\item kontinuierliches Medium 
		\item Datengröße mittel
		\item Kompression: MP3
		\item Schwingung 
		\item Intensität
		\item Abtastrate/Bitrate
	\end{itemize}
	
	\textbf{Video}
	\begin{itemize}
		\item kontinuierliches Medium 
		\item Datengröße sehr groß
		\item Kompression: MPEG, H.264
		\item Abfolge von Frames 
		\item Framerate
	\end{itemize}
